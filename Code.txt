GLOBAL _start

SECTION .data

    msg DB "Hello, World!" 0x0A

SECTION .text

_start:
    MOV ECX 0x00
beg:
    PUSH RCX
    MOV EDX 0x0E
    MOV ECX msg
    MOV EBX 0x01
    MOV EAX 0x04
    INT 0x80
    POP RCX
    INC RCX
    CMP RCX 0x0A
    JNE beg
    MOV EBX 0x00
    MOV EAX 0x01
    INT 0x80
